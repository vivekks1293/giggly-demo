Giggly Text Area

The Giggly Textarea is a multi-line input component designed for capturing longer text inputs in a playful yet professional manner. It is part of the Giggly UI library, offering a seamless user experience with auto-resizing capabilities and easy integration with Angular applications. 

<giggly-textarea name="feedback" label="Share your feedback" placeholder="Type something..."
                    [(ngModel)]="feedbackModel.message" #feedbackField="ngModel" [required]="true"
                    errorMessage="Feedback is required!" [rows]="3" 
                    [giggleLevel]="'medium'"></giggly-textarea>

<giggly-textarea name="funExperience" label="Tell us about a fun experience"
                    placeholder="Share your best memory..." [(ngModel)]="feedbackModel.funExperience"
                    #funField="ngModel" [required]="true" errorMessage="We need at least one fun story!" [rows]="4"
                    [giggleLevel]="'high'" [giggleEmojis]="['ü•≥', '‚ú®', 'üéâ', 'üéä', 'üëè']"
                     [confettiOnComplete]="true"
                    (giggleTriggered)="onGiggle()" (valueChange)="onValueChange($event)"></giggly-textarea>

<giggly-textarea name="suggestions" label="Any suggestions for improvement?"
                    placeholder="Optional but appreciated..." [(ngModel)]="feedbackModel.suggestions"
                    [giggleLevel]="'low'" [giggleSounds]="false" [rows]="2"></giggly-textarea>

                        
feedbackModel = {
    message: '',
    funExperience: '',
    suggestions: ''
  };

onGiggle() {
  console.log('Giggle triggered!');
}

onValueChange(value: string) {
  console.log('Value changed:', value);
}
