Giggly Text Box

The Giggly TextBox is an Angular input component that combines a floating label, placeholder and built-in validation (required, regex patterns) with customizable error messages. Its playful animations can be tuned via the gigglyLevel (“low” | “medium” | “high”), and it binds seamlessly using [(ngModel)] for easy form integration.

Personal Information

<giggly-text-box label="Your Name" placeholder="Enter your full name" [(ngModel)]="contactModel.name"
                    name="name" #name="ngModel" [required]="true" 
                    errorMessage="Please tell us your name"
                    gigglyLevel="medium">
</giggly-text-box>

<giggly-text-box label="Email Address" placeholder="Enter your email" [(ngModel)]="contactModel.email"
                    name="email" #email="ngModel" [required]="true" type="email"
                    errorMessage="Please enter a valid email address" 
                    pattern="[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$"
                    gigglyLevel="medium">
</giggly-text-box>

<giggly-text-box label="Phone Number (Optional)" placeholder="Enter your phone number"
                    [(ngModel)]="contactModel.phone" name="phone" #phone="ngModel" type="tel" 
                    pattern="^[0-9]{10}$"
                    errorMessage="Please enter a valid 10-digit phone number" gigglyLevel="medium"
                    (valueChange)="onPhoneChange($event)">
</giggly-text-box>

<giggly-text-box label="Your Message" placeholder="What would you like to tell us?"
                    [(ngModel)]="contactModel.message" name="message" #message="ngModel" 
                    [required]="true"
                    [showCharCount]="true" [maxLength]="300" errorMessage="Please enter your message"
                    gigglyLevel="high">
</giggly-text-box>
        
                        
interface ContactModel {
  name: string;
  email: string;
  phone: string;
  message: string;
}

contactModel: ContactModel = {
  name: '',
  email: '',
  phone: '',
  message: ''
};

onPhoneChange(value: string): void {
  console.log('Phone changed:', value);
}
