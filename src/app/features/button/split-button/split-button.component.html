Giggly Split Button

The Split Button is a dynamic UI component designed to combine a primary action button with a dropdown of additional options. Itâ€™s part of the Giggly UI library, bringing playful and functional design elements to your Angular applications.

Button Preview

  <giggly-split-button
  (onClick)="onSave()"
  (onOptionSelect)="onOptionSelected($event)"
  variant="primary"
  label="Save"
  [options]="[
    { label: 'Save as Draft', action: 'draft' },
    { label: 'Save and Publish', action: 'publish' },
    { label: 'Save as Template', action: 'template' }
  ]">
</giggly-split-button>
                    
onSave() {
  console.log('Default save action');  
}

onOptionSelected(action: string) {
  console.log('Selected action:', action);  
  // Handle the specific action  
  switch(action) {
    case 'draft':
      // Save as draft logic  
      break;
    case 'publish':
      // Save and publish logic  
      break;
    case 'template':
      // Save as template logic  
      break;
  }
